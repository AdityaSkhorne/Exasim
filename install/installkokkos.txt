# Build Kokkos Serial Library
cd Exasim/kokkos
mkdir buildserial
cd buildserial
cmake .. -DCMAKE_INSTALL_PREFIX=../buildserial
make install   

# Build Kokkos CUDA Library
cd Exasim/kokkos
mkdir buildcuda
cd buildcuda
cmake .. -DCMAKE_CXX_COMPILER=clang++ -DKokkos_ENABLE_CUDA=ON -DCMAKE_INSTALL_PREFIX=../buildcuda
make install
